function doGet(e)  { return handleRequest(e); }

function handleRequest(e) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  let payload = {};
  if (e && e.parameter) payload = { ...e.parameter };
  if (e && e.postData && e.postData.contents) {
    try { payload = { ...payload, ...JSON.parse(e.postData.contents) }; } catch (err) {}
  }
  const action = (payload.action || "").toLowerCase().trim();
  if (!action) return jsonResponse({ status: "error", message: "No action provided" });

  try {
    if (action === "login")             return loginUser(ss, payload);
    if (action === "getadmindata")      return getAdminData(ss, payload);
    if (action === "attendance")        return markAttendance(ss, payload);
    if (action === "gettodayattendance") return getTodayAttendance(ss, payload);
    if (action === "uploadorder")       return uploadOrder(ss, payload);
    if (action === "getorder")          return getOrder(ss, payload);
    if (action === "logalertv2")        return logAlertV2(ss, payload);
    if (action === "uploadmatrixdata")  return uploadMatrixData(ss, payload);
    if (action === "getmatrixdata")     return getMatrixData(ss);
    if (action === "getalertlogs")      return getAlertLogs(ss);
    if (action === "getregions")        return getRegions(ss);
    if (action === "resetattendance")   return resetAttendance(ss, payload);
    if (action === "syncuser")          return syncUser(ss, payload); // ✅ NEW: One-way sync from PWA
    if (action === "getstaffdashboard") return getStaffDashboard(ss, payload);
    return jsonResponse({ status: "error", message: "Invalid action: " + action });
  } catch (err) {
    return jsonResponse({ status: "error", message: err.toString() });
  }
}

/**
 * GET STAFF DASHBOARD — Aggregated view of coverage across stores
 */
function getStaffDashboard(ss, params) {
  const usersSheet = ss.getSheetByName("Users");
  const attendanceSheet = ss.getSheetByName("Attendance");
  
  if (!usersSheet) return jsonResponse({ status: "error", message: "Users sheet missing" });
  
  const userData = usersSheet.getDataRange().getValues();
  const usersH = userData.shift().map(h => String(h).toLowerCase().trim());
  const ui=usersH.indexOf("username"), ri=usersH.indexOf("role"), sti=usersH.indexOf("storeid"),
        ni=usersH.indexOf("name"), ei=usersH.indexOf("empid"), ssi=usersH.indexOf("shiftstart"),
        shi=usersH.indexOf("shifthours"), woi=usersH.indexOf("weekoffday"), si=usersH.indexOf("status");

  // Today's attendance (last 2000 rows for performance)
  let attRecords = [];
  if (attendanceSheet) {
    const lastRow = attendanceSheet.getLastRow();
    if (lastRow > 1) {
      const rowCount = Math.min(2000, lastRow - 1);
      attRecords = attendanceSheet.getRange(lastRow - rowCount + 1, 1, rowCount, 6).getValues();
    }
  }

  const todayStr = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyy-MM-dd");
  const currentHour = new Date().getHours();
  const currentDay = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "EEEE"); // e.g. "Friday"

  // Process users into timeline
  const staffList = userData.map(row => {
    const empId = String(row[ei]).trim();
    const storeId = String(row[sti]).trim();
    const role = String(row[ri]).toLowerCase().trim();
    const status = String(row[si] || "Active").toLowerCase().trim();
    
    if (!empId || status !== "active") return null;
    if (role !== "picker" && role !== "driver") return null;

    // Filter by attendance
    const userAtt = attRecords.filter(r => String(r[0]).trim() === empId);
    let punchedIn = false;
    let punchedOut = false;
    userAtt.forEach(r => {
      try {
        const d = Utilities.formatDate(new Date(r[4]), Session.getScriptTimeZone(), "yyyy-MM-dd");
        if (d === todayStr) {
          if (r[3] === "In") punchedIn = true;
          if (r[3] === "Out") punchedOut = true;
        }
      } catch(e) {}
    });

    const shiftStart = Number(row[ssi] || 6);
    const shiftHours = Number(row[shi] || 8);
    const weekOffDay = String(row[woi] || "").trim();
    const isWeekOff = weekOffDay.toLowerCase() === currentDay.toLowerCase();
    
    const shiftEnd = shiftStart + shiftHours;
    const isActive = punchedIn && !punchedOut;

    return {
      empId,
      name: row[ni] || row[ui],
      role,
      storeId,
      shiftStart,
      shiftHours,
      isActive,
      isWeekOff,
      punchedIn
    };
  }).filter(x => x !== null);

  // Group by store
  const storeMap = {};
  staffList.forEach(s => {
    if (!storeMap[s.storeId]) {
      storeMap[s.storeId] = {
        storeId: s.storeId,
        totalStaff: 0,
        activeNow: 0,
        weekOff: 0,
        notStarted: 0,
        staffTimeline: []
      };
    }
    const sm = storeMap[s.storeId];
    sm.totalStaff++;
    if (s.isActive) sm.activeNow++;
    else if (s.isWeekOff) sm.weekOff++;
    else if (!s.punchedIn) sm.notStarted++;
    sm.staffTimeline.push(s);
  });

  const storeBreakdown = Object.values(storeMap);

  // Hourly breakdown
  const hourlyBreakdown = [];
  for (let h = 0; h <= 23; h++) {
    const label = h === 0 ? "12 AM" : h < 12 ? (h + " AM") : (h === 12 ? "12 PM" : (h - 12) + " PM");
    let active = 0, weekOff = 0, notStarted = 0;
    
    staffList.forEach(s => {
      let shiftStart = s.shiftStart;
      let shiftHours = s.shiftHours;
      let shiftEnd = shiftStart + shiftHours;
      
      // Handle shifts that cross midnight
      let isInShift = false;
      if (shiftEnd > 24) {
        // e.g. 22 to 6 AM (shiftEnd = 30)
        isInShift = (h >= shiftStart) || (h < (shiftEnd - 24));
      } else {
        isInShift = (h >= shiftStart && h < shiftEnd);
      }

      if (isInShift) {
        if (s.isActive) active++;
        else if (s.isWeekOff) weekOff++;
        else if (!s.punchedIn) notStarted++;
      }
    });

    hourlyBreakdown.push({ hour: h, label, active, weekOff, notStarted });
  }

  const summary = {
    totalStores: storeBreakdown.length,
    totalStaff: staffList.length,
    activeNow: staffList.filter(s => s.isActive).length,
    weekOff: staffList.filter(s => s.isWeekOff).length,
    notLoggedIn: staffList.filter(s => !s.punchedIn && !s.isWeekOff).length
  };

  return jsonResponse({
    status: "success",
    data: {
      summary,
      hourlyBreakdown,
      storeBreakdown,
      lastUpdated: new Date().toISOString()
    }
  });
}

/**
 * SYNC USER — ✅ NEW: Keeps "Users" sheet updated when changes happen in PWA Admin
 */
function syncUser(ss, params) {
  const sheet = ss.getSheetByName("Users") || ss.insertSheet("Users");
  const data = sheet.getDataRange().getValues();
  
  if (data.length === 0) {
    const defaultHeaders = ["username", "password", "role", "storeId", "name", "status", "empId", "ProfileImage", "region", "shiftStart", "shiftHours", "weekOffDay"];
    sheet.appendRow(defaultHeaders);
    data.push(defaultHeaders);
  }

  const headers = data[0].map(h => String(h).toLowerCase().trim());
  const ei = headers.indexOf("empid");
  const actionType = String(params.syncAction || "upsert").toLowerCase();
  const empId = String(params.empId || "").trim();
  
  if (!empId) return jsonResponse({ status: "error", message: "empId required for sync" });

  let foundRow = -1;
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][ei]).trim() === empId) {
      foundRow = i + 1;
      break;
    }
  }

  if (actionType === "delete") {
    if (foundRow !== -1) {
      sheet.deleteRow(foundRow);
      return jsonResponse({ status: "success", message: "User deleted from sheet" });
    }
    return jsonResponse({ status: "success", message: "User not in sheet" });
  }

  // Upsert logic
  const rowData = headers.map(h => {
    const key = h.toLowerCase().trim();
    if (key === "status") return params.status || params.Status || "Active";
    if (key === "profileimage") return params.profileImage || params.profileimage || params.ProfileImage || "";
    if (key === "storeid") return params.storeId || params.storeid || params.StoreID || "";
    if (key === "empid") return params.empId || params.empid || params.EmpId || "";
    
    // Fallback search
    for (let pKey in params) {
      if (pKey.toLowerCase() === key) return params[pKey];
    }
    return "";
  });

  if (foundRow !== -1) {
    sheet.getRange(foundRow, 1, 1, rowData.length).setValues([rowData]);
    return jsonResponse({ status: "success", message: "User updated in sheet" });
  } else {
    sheet.appendRow(rowData);
    return jsonResponse({ status: "success", message: "User appended to sheet" });
  }
}

function loginUser(ss, params) {
  const sheet = ss.getSheetByName("Users");
  if (!sheet) return jsonResponse({ status: "error", message: "Users sheet missing" });
  const data = sheet.getDataRange().getValues();
  const h = data[0].map(x => String(x).toLowerCase().trim());
  const ui=h.indexOf("username"),pi=h.indexOf("password"),si=h.indexOf("status"),
        ri=h.indexOf("role"),sti=h.indexOf("storeid"),ni=h.indexOf("name"),
        ei=h.indexOf("empid"),regi=h.indexOf("region"), pri=h.indexOf("profileimage"),
        ssi=h.indexOf("shiftstart"), shi=h.indexOf("shifthours"), woi=h.indexOf("weekoffday");
  const username=(params.username||"").trim(), password=(params.password||"").trim();
  for (let i=1;i<data.length;i++) {
    if (String(data[i][ui]).trim()===username && String(data[i][pi]).trim()===password) {
      const userStatus = String(data[i][si] || "Active").toLowerCase().trim();
      if (userStatus !== "active")
        return jsonResponse({ status:"error", message:"Account status is " + userStatus });
        
      return jsonResponse({ status:"success", user:{
        username: String(data[i][ui]), 
        role: data[i][ri]||"user", 
        storeId: String(data[i][sti]||""),
        name: data[i][ni]||username, 
        empId: String(data[i][ei]||""), 
        region: data[i][regi]||"",
        status: data[i][si] || "Active",
        profileImage: pri !== -1 ? data[i][pri] : "",
        shiftStart: ssi !== -1 ? Number(data[i][ssi] || 6) : 6,
        shiftHours: shi !== -1 ? Number(data[i][shi] || 8) : 8,
        weekOffDay: woi !== -1 ? String(data[i][woi] || "") : ""
      }});
    }
  }
  return jsonResponse({ status:"error", message:"Invalid credentials" });
}

// ─────────────────────────────────────────────
function getAdminData(ss, params) {
  const role = String(params.role || "").toLowerCase().trim();
  const filterStoreId = String(params.storeId || "").trim().toLowerCase();
  const isPrivileged = role === 'admin' || role === 'supervisor';

  const usersSheet = ss.getSheetByName("Users");
  const attendanceSheet = ss.getSheetByName("Attendance");
  const ordersSheet = ss.getSheetByName("Orders");
  const regionsSheet = ss.getSheetByName("Regions");
  const result = { users: [], attendance: [], orders: [], regions: [] };

  if (usersSheet) {
    const data = usersSheet.getDataRange().getValues();
    const rawHeaders = data.shift();
    const headers = rawHeaders.map(h => {
      const s = String(h).toLowerCase().trim();
      if (s === "empid") return "empId";
      if (s === "storeid") return "storeId";
      if (s === "profileimage") return "profileImage";
      if (s === "shiftstart") return "shiftStart";
      if (s === "shifthours") return "shiftHours";
      if (s === "weekoffday") return "weekOffDay";
      return s;
    });

    result.users = data.map(r => {
      let obj = {};
      headers.forEach((h, i) => { if (h.toLowerCase() !== "password") obj[h] = r[i]; });
      return obj;
    });
  }

  if (regionsSheet) {
    const data = regionsSheet.getDataRange().getValues();
    if (data.length > 1) {
      const h = data[0].map(x => String(x).toLowerCase().trim());
      const si=h.indexOf("storeid"), ri=h.indexOf("region");
      result.regions = data.slice(1).map(r => ({
        storeId: String(r[si]||"").trim(), region: String(r[ri]||"").trim()
      }));
    }
  }

  if (attendanceSheet) {
    const lastRow = attendanceSheet.getLastRow();
    if (lastRow > 1) {
      const rowCount = Math.min(1000, lastRow - 1);
      const startRow = lastRow - rowCount + 1;
      const data = attendanceSheet.getRange(startRow, 1, rowCount, 6).getValues();
      result.attendance = data.map(r => ({
        empId:r[0], name:r[1], storeId:r[2], type:r[3], timestamp:r[4], imageUrl:r[5]
      })).reverse();
    }
  }

  if (ordersSheet) {
    const lastRow = ordersSheet.getLastRow();
    if (lastRow > 1) {
      const rowCount = Math.min(1000, lastRow - 1);
      const startRow = lastRow - rowCount + 1;
      // ✅ Read 7 columns (was 6)
      const data = ordersSheet.getRange(startRow, 1, rowCount, 7).getValues();
      result.orders = data.map(r => ({
        orderId:r[0], storeId:r[1], pickerName:r[2],
        imageUrl:r[3], timestamp:r[4], uploadedBy:r[5],
        allImages: r[6] || r[3]  // ✅ Column 7 = all images comma-separated
      })).reverse();

      // ✅ FIXED BUG 7: Server-side filter for non-admin roles
      if (!isPrivileged && filterStoreId && filterStoreId !== 'all') {
        result.orders = result.orders.filter(o =>
          String(o.storeId||"").trim().toLowerCase() === filterStoreId
        );
      }
    }
  }

  return jsonResponse({ status: "success", data: result });
}

function getTodayAttendance(ss, params) {
  const sheet = ss.getSheetByName("Attendance");
  if (!sheet) return jsonResponse({ status:"success", data:[] });
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return jsonResponse({ status:"success", data:[] });
  const data = sheet.getRange(Math.max(2, lastRow-500), 1, Math.min(500, lastRow-1), 6).getValues();
  const today = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyy-MM-dd");
  const filtered = data.filter(r => {
    try {
      return Utilities.formatDate(new Date(r[4]), Session.getScriptTimeZone(), "yyyy-MM-dd") === today
        && (!params.empId || String(r[0]).trim() === String(params.empId).trim());
    } catch(e) { return false; }
  }).map(r => ({ empId:r[0], name:r[1], storeId:r[2], type:r[3], timestamp:r[4], imageUrl:r[5] }));
  return jsonResponse({ status:"success", data:filtered });
}

function resetAttendance(ss, params) {
  const sheet = ss.getSheetByName("Attendance");
  if (!sheet) return jsonResponse({ status:"error", message:"Attendance sheet missing" });
  const empId = String(params.empId || "").trim();
  const filterDate = String(params.date || "").trim();
  if (!empId || !filterDate) return jsonResponse({ status:"error", message:"empId and date required" });
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return jsonResponse({ status:"success", deleted:0 });
  const data = sheet.getRange(2, 1, lastRow-1, 6).getValues();
  let deletedCount = 0;
  for (let i = data.length-1; i >= 0; i--) {
    const rowEmpId = String(data[i][0]||"").trim();
    let rowDate = "";
    try { rowDate = Utilities.formatDate(new Date(data[i][4]), Session.getScriptTimeZone(), "yyyy-MM-dd"); }
    catch(e) { continue; }
    if (rowEmpId === empId && rowDate === filterDate) {
      sheet.deleteRow(i + 2);
      deletedCount++;
    }
  }
  return jsonResponse({ status:"success", deleted:deletedCount });
}

// ─────────────────────────────────────────────
function uploadOrder(ss, params) {
  const sheet = ss.getSheetByName("Orders") || ss.getSheetByName("DailyOrders");
  if (!sheet) return jsonResponse({ status:"error", message:"Orders sheet missing" });

  const orderId = params.orderId.toString().startsWith("Lulu-")
    ? params.orderId : "Lulu-" + params.orderId;

  // ✅ DUPLICATE CHECK
  const lastRow = sheet.getLastRow();
  if (lastRow > 1) {
    const existing = sheet.getRange(2, 1, lastRow-1, 6).getValues();
    for (let i=0; i<existing.length; i++) {
      if (String(existing[i][0]).trim().toLowerCase() === orderId.toLowerCase()) {
        return jsonResponse({ status:"duplicate", existing:{
          orderId:existing[i][0], storeId:existing[i][1], pickerName:existing[i][2],
          imageUrl:existing[i][3], timestamp:existing[i][4], uploadedBy:existing[i][5]
        }});
      }
    }
  }

  let imageUrls = [];
  if (params.image) {
    const images = String(params.image).split("|||");
    images.forEach((img, index) => {
      if (!img) return;
      const url = uploadImageToDrive(
        img, "Order_"+orderId+"_"+(index+1)+".jpg",
        ORDERS_ROOT_ID, [params.storeId]
      );
      if (url) imageUrls.push(url);
    });
  }

  const primaryImage = imageUrls.length > 0 ? imageUrls[0] : "";
  sheet.appendRow([
    orderId, params.storeId, params.pickerName,
    primaryImage, new Date(), params.uploadedBy,
    imageUrls.join(",")
  ]);
  return jsonResponse({ status:"success", imageUrl:primaryImage, allImages:imageUrls });
}

// ─────────────────────────────────────────────
function logAlertV2(ss, params) {
  const sheet = getOrCreateSheet(ss, "AlertLogs");
  const headers = ["id","timestamp","orderId","eventType","storeId","userId",
    "notificationTime","storeStaffName","status","escalation","managerName",
    "statusTrigger","managerStatus","orderCreatedAt", "bucket"];
  
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(headers);
  } else {
    // Check if we need to add the 'id' column or 'bucket' column if they are missing
    const existingHeaders = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0].map(h => String(h).toLowerCase().trim());
    if (!existingHeaders.includes("id")) {
      sheet.insertColumnBefore(1);
      sheet.getRange(1, 1).setValue("id");
    }
    if (!existingHeaders.includes("bucket")) {
      sheet.insertColumnAfter(sheet.getLastColumn());
      sheet.getRange(1, sheet.getLastColumn()).setValue("bucket");
    }
  }

  // Reload headers to be sure about positions
  const finalHeaders = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0].map(h => String(h).toLowerCase().trim());
  const row = finalHeaders.map(h => {
    // Match param names (case-insensitive) to headers
    const key = Object.keys(params).find(k => k.toLowerCase() === h);
    return key ? params[key] : "";
  });
  
  sheet.appendRow(row);
  return jsonResponse({ status:"success" });
}

function getAlertLogs(ss) {
  const sheet = ss.getSheetByName("AlertLogs");
  if (!sheet) return jsonResponse({ status:"success", data:[] });
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return jsonResponse({ status:"success", data:[] });
  
  const MAX_ROWS = 1500;
  const startRow = Math.max(2, lastRow - MAX_ROWS + 1);
  const numRows = lastRow - startRow + 1;
  const lastCol = sheet.getLastColumn();
  const headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0].map(h => String(h).toLowerCase().trim());
  const data = sheet.getRange(startRow, 1, numRows, lastCol).getValues();
  
  const rows = data.map(r => {
    let obj = {};
    headers.forEach((h,i) => { obj[h] = r[i]; });
    return obj;
  });
  return jsonResponse({ status:"success", data: rows.reverse() });
}

function uploadMatrixData(ss, payload) {
  const lock = LockService.getScriptLock();
  if (!lock.tryLock(20000)) return jsonResponse({ status:"error", message:"Busy, retry" });
  try {
    const jsonStr = typeof payload.data==="string" ? payload.data : JSON.stringify(payload.data);
    const liveSheet = ss.getSheetByName("MatrixLive") || ss.insertSheet("MatrixLive");
    storeAcrossRows(liveSheet, jsonStr, payload.timestamp);
    SpreadsheetApp.flush();
    return jsonResponse({ status:"success" });
  } finally { lock.releaseLock(); }
}

function getMatrixData(ss) {
  const sheet = ss.getSheetByName("MatrixLive");
  if (!sheet) return jsonResponse({ status:"success", data:{quick:[],schedule:[]}, timestamp:"" });
  try {
    const timestamp = sheet.getRange("B1").getValue();
    const rowCount = parseInt(sheet.getRange("B2").getValue() || 1);
    const merged = sheet.getRange(1, 1, rowCount, 1).getValues().flat().join("");
    return jsonResponse({ status:"success", data: merged ? JSON.parse(merged) : {quick:[],schedule:[]}, timestamp });
  } catch (e) { return jsonResponse({ status:"error", message:e.toString() }); }
}

function storeAcrossRows(sheet, jsonStr, timestamp) {
  const LIMIT = 45000;
  sheet.clearContents();
  const rows = [];
  for (let i=0; i<jsonStr.length; i+=LIMIT) rows.push([jsonStr.slice(i,i+LIMIT)]);
  if (rows.length > 0) sheet.getRange(1,1,rows.length,1).setValues(rows);
  sheet.getRange(1,2).setValue(timestamp || new Date().toISOString());
  sheet.getRange(2,2).setValue(rows.length);
  SpreadsheetApp.flush();
}

// ─────────────────────────────────────────────
// IMAGE UPLOAD TO GOOGLE DRIVE
// ─────────────────────────────────────────────
function uploadImageToDrive(base64, fileName, rootFolderId, pathSegments) {
  try {
    if (!base64 || !base64.includes(",")) return "";
    const bytes = Utilities.base64Decode(base64.split(",")[1]);
    const blob = Utilities.newBlob(bytes, "image/jpeg", fileName);
    let currentFolder = DriveApp.getFolderById(rootFolderId);
    if (pathSegments && pathSegments.length > 0) {
      pathSegments.forEach(segment => {
        if (!segment) return;
        const name = String(segment).trim();
        const folders = currentFolder.getFoldersByName(name);
        currentFolder = folders.hasNext()
          ? folders.next()
          : currentFolder.createFolder(name);
        currentFolder.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      });
    }
    const file = currentFolder.createFile(blob);
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    return "https://lh3.googleusercontent.com/d/" + file.getId();
  } catch (e) {
    Logger.log("Upload Failed: " + e.toString());
    return "";
  }
}
